package main

import "fmt"

// fibonacci 函数会返回一个返回 int 的函数。
func fibonacci() func() int {
    sum := 0
    a := 0
    b := 1
    return func() int {
        sum =  a + b
        a = b
        b = sum
        return sum
    }
}

func main() {
    f := fibonacci()
    fmt.Println(f())
    for i := 0; i < 10; i++ {
        fmt.Println(f())
    }
}
